// Definindo variáveis
let numeroUm = 1; // Define a variável numeroUm com o valor numérico 1
let stringUm = '1'; // Define a variável stringUm com o valor string '1'
let numeroTrinta = 30; // Define a variável numeroTrinta com o valor numérico 30
let stringTrinta = '30'; // Define a variável stringTrinta com o valor string '30'
let numeroDez = 10; // Define a variável numeroDez com o valor numérico 10
let stringDez = '10'; // Define a variável stringDez com o valor string '10'

// Comparação entre numeroUm e stringUm
if (numeroUm == stringUm) {
  console.log('As variáveis numeroUm e stringUm têm o mesmo valor, mas tipos diferentes'); // Se numeroUm e stringUm têm o mesmo valor, imprime esta mensagem
} else {
  console.log('As variáveis numeroUm e stringUm não têm o mesmo valor'); // Se numeroUm e stringUm não têm o mesmo valor, imprime esta mensagem
}

// Comparação entre numeroTrinta e stringTrinta
if (numeroTrinta === stringTrinta) {
  console.log('As variáveis numeroTrinta e stringTrinta têm o mesmo valor e mesmo tipo'); // Se numeroTrinta e stringTrinta têm o mesmo valor e tipo, imprime esta mensagem
} else {
  console.log('As variáveis numeroTrinta e stringTrinta não têm o mesmo tipo'); // Se numeroTrinta e stringTrinta não têm o mesmo tipo, imprime esta mensagem
}

// Comparação entre numeroDez e stringDez
if (numeroDez == stringDez) {
  console.log('As variáveis numeroDez e stringDez têm o mesmo valor, mas tipos diferentes'); // Se numeroDez e stringDez têm o mesmo valor, imprime esta mensagem
} else {
  console.log('As variáveis numeroDez e stringDez não têm o mesmo valor'); // Se numeroDez e stringDez não têm o mesmo valor, imprime esta mensagem
}
